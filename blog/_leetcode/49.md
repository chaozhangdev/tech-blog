```js
/**
 * @param {string[]} strs
 * @return {string[][]}
 */
var groupAnagrams = function(strs) {
  let map = {};
  for (let str of strs) {
    let strSorted = str
      .split("")
      .sort()
      .join("");
    if (!map[strSorted]) {
      map[strSorted] = [str];
    } else {
      map[strSorted].push(str);
    }
  }
  let result = [];
  for (let el of Object.values(map)) {
    result.push(el);
  }
  return result;
};
```
